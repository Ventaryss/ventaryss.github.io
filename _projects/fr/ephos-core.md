<!-- Description -->
<section class="py-20">
    <div class="max-w-5xl mx-auto px-4">
        <div class="bg-white dark:bg-dark-navbar rounded-2xl shadow-2xl p-8 md:p-12">
            <h2 class="text-3xl font-bold mb-6 text-portfolio-violet">
                <i class="fas fa-info-circle mr-3"></i>À propos du projet
            </h2>
            <p class="text-lg text-gray-700 dark:text-gray-300 mb-6">
                <strong>EPHOS Core</strong> est une plateforme ERP complète développée pour gérer l'ensemble des opérations d'une entreprise de cybersécurité : gestion de clients, projets, audits, finances, communications et intégrations externes.
            </p>
            <p class="text-lg text-gray-700 dark:text-gray-300 mb-6">
                Conçu avec une <strong>architecture modulaire</strong> basée sur les bounded contexts (DDD), EPHOS Core offre une flexibilité maximale pour s'adapter aux besoins spécifiques de chaque organisation tout en maintenant une base de code maintenable et évolutive.
            </p>
            <p class="text-lg text-gray-700 dark:text-gray-300 mb-6">
                La plateforme intègre <strong>ZENITH</strong>, un assistant IA conversationnel qui aide les utilisateurs dans leurs tâches quotidiennes, et propose plus de <strong>29 intégrations</strong> avec des services externes (Stripe, Slack, Google Drive, etc.).
            </p>
        </div>
    </div>
</section>
